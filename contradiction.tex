\begin{haskellcode}
data Perhaps a = Unknown | Known a | Contradiction
\end{haskellcode}

\pnl

\begin{haskellcode}
instance Eq a => BoundedJoinSemiLattice (Perhaps a) where

  bottom = Unknown

  (\/) Unknown x           = x
  (\/) x       Unknown     = x
  (\/) Contradiction _     = Contradiction
  (\/) _     Contradiction = Contradiction
  (\/) (Known a) (Known b) =
    if a == b
      then Known a
      else Contradiction
\end{haskellcode}
